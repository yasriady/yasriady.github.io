#include <WiFi.h>
#include <WiFiClientSecure.h>
#include <HTTPClient.h>

// ================= CONFIG =================
const char* WIFI_SSID     = "Wokwi-GUEST";
const char* WIFI_PASSWORD = "";

const char* API_URL   = "https://iotweb.apache.web.id/api/save-data";
const char* API_TOKEN = "f62504400873ef044d68126c13423223";

const char* DEVICE_ID = "esp32-wokwi-02";

const int TEMP_PIN = 34;
// ==========================================

void setup() {
  Serial.begin(115200);

  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);

  Serial.print("Menghubungkan WiFi");

  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }

  Serial.println();
  Serial.println("WiFi Connected");
}

float readTemperature() {
  int raw = analogRead(TEMP_PIN);

  float voltage = raw * 3.3 / 4095.0;
  float tempC = (voltage - 0.5) * 100.0;

  // fallback simulasi Wokwi
  if (tempC < -40 || tempC > 85) {
    tempC = 25 + random(0, 100) / 10.0;
  }

  return tempC;
}

void loop() {

  float temp = readTemperature();

  Serial.print("Suhu: ");
  Serial.println(temp);

  String json =
    "{\"temperature\":" +
    String(temp, 2) +
    ",\"device_id\":\"" +
    DEVICE_ID +
    "\"}";

  WiFiClientSecure client;
  client.setInsecure();

  HTTPClient http;

  http.begin(client, API_URL);

  http.addHeader("Content-Type", "application/json");
  http.addHeader("Authorization", String("Bearer ") + API_TOKEN);

  int httpCode = http.POST(json);

  Serial.print("HTTP Code: ");
  Serial.println(httpCode);

  if (httpCode > 0) {
    Serial.println(http.getString());
  }

  http.end();

  delay(3000);
}
